package Connect;


import com.mongodb.DB; 
import com.mongodb.MongoClient; 
import com.mongodb.BasicDBObject;
import com.mongodb.DBCollection;

public class MakeConnection { 
	public static void main(String[] args) {
		try { 
			// code to create the connection
			MongoClient mongoClient = new MongoClient("localhost", 27017); 
			// code to connect to the database
			DB db = mongoClient.getDB("T3BATCH");

			// get the User collection from DB
			DBCollection userCollection = db.getCollection("Abhinav");
			BasicDBObject bO = new BasicDBObject();

			// adding custom _id
			
			bO.put("name", "AMRUTVAHINI");
		
			// inserting data into collection
			userCollection.insert(bO);
		} 
		catch(Exception e) { 
			e.printStackTrace(); 
		} 
	}
}

import com.mongodb.client.MongoClients;
import com.mongodb.client.MongoClient;
import com.mongodb.client.MongoDatabase;
import com.mongodb.client.MongoCollection;
import org.bson.Document;

public class MakeConnection {
    public static void main(String[] args) {
        try {
            // Create connection
            MongoClient mongoClient = MongoClients.create("mongodb://localhost:27017");

            // Connect to database
            MongoDatabase database = mongoClient.getDatabase("T3BATCH");

            // Get the collection
            MongoCollection<Document> collection = database.getCollection("Abhinav");

            // Create a new document
            Document doc = new Document("name", "AMRUTVAHINI");

            // Insert into collection
            collection.insertOne(doc);

            System.out.println("Document inserted successfully!");
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
